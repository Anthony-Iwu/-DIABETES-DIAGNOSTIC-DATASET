import pandas as pd
import matplotlib.pyplot as plt

# Load dataset
df = pd.read_csv('diabetes.csv')

# Calculate proportion of zeros in SkinThickness and Insulin for each group
missing_data = df.groupby('Outcome').agg({
    'SkinThickness': lambda x: (x == 0).mean(),  # Mean of zeros = proportion missing
    'Insulin': lambda x: (x == 0).mean()
}).reset_index()

# Melt data for plotting
missing_melted = pd.melt(
    missing_data,
    id_vars='Outcome',
    value_vars=['SkinThickness', 'Insulin'],
    var_name='Feature',
    value_name='Proportion Missing'
)

# Map Outcome to labels for clarity
missing_melted['Outcome'] = missing_melted['Outcome'].map({0: 'Non-Diabetic', 1: 'Diabetic'})

# Plot
plt.figure(figsize=(10, 6))
for i, feature in enumerate(['SkinThickness', 'Insulin']):
    subset = missing_melted[missing_melted['Feature'] == feature]
    plt.bar(
        [i - 0.2, i + 0.2],  # Offset bars for grouping
        subset['Proportion Missing'],
        width=0.4,
        label=feature if i == 0 else '',
        color=['#1f77b4', '#ff7f0e']
    )

# Formatting
plt.xticks([0, 1], ['SkinThickness', 'Insulin'])
plt.ylabel('Proportion of Zeros (Missing)')
plt.title('Missing Data (Zeros) by Diabetes Status')
plt.legend(['Non-Diabetic', 'Diabetic'])
plt.grid(axis='y', alpha=0.3)
plt.savefig("status.png")
plt.show()

What thresholds for glucose or BMI maximize sensitivity/specificity in a diabetes screening tool?¶

import pandas as pd
import numpy as np  # Ensure numpy is imported
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import roc_curve

# Load data
df = pd.read_csv('diabetes.csv')

# Prepare features and target
X = df[['Glucose', 'BMI', 'Age']]
y = df['Outcome']

# Split data into train/test sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Train a model (e.g., logistic regression)
model = LogisticRegression()
model.fit(X_train, y_train)

# Get predicted probabilities for the positive class
y_scores = model.predict_proba(X_test)[:, 1]  # Use [:,1] for class=1 probabilities

# Compute ROC curve
fpr, tpr, thresholds = roc_curve(y_test, y_scores)

# Calculate optimal threshold using Youden's J statistic
youden_j = tpr - fpr
optimal_idx = np.argmax(youden_j)  # Requires numpy
optimal_threshold = thresholds[optimal_idx]

# Plot sensitivity/specificity vs. threshold
plt.figure()
plt.plot(thresholds, tpr, label='Sensitivity (TPR)')
plt.plot(thresholds, 1 - fpr, label='Specificity (TNR)')
plt.axvline(optimal_threshold, color='red', linestyle='--', label=f'Optimal Threshold: {optimal_threshold:.2f}')
plt.xlabel('Threshold')
plt.ylabel('Rate')
plt.legend()
plt.title('Sensitivity/Specificity Tradeoff')
plt.savefig("diabetes sceening tool.png")
plt.show()

import pandas as pd
from sklearn.linear_model import LogisticRegression
from sklearn.feature_selection import RFE  # Import RFE

# Load data
df = pd.read_csv('diabetes.csv')
X = df.drop('Outcome', axis=1)
y = df['Outcome']

# Initialize the model
model = LogisticRegression(max_iter=1000)

# Recursive Feature Elimination
accuracies = []
for n in range(1, 9):
    rfe = RFE(model, n_features_to_select=n)  # Now RFE is defined
    X_rfe = rfe.fit_transform(X, y)
    model.fit(X_rfe, y)
    accuracies.append(model.score(X_rfe, y))

print(accuracies)
# Plot
plt.plot(range(1, 9), accuracies, marker='o')
plt.xlabel('Number of Features Selected')
plt.ylabel('Accuracy')
plt.title('Accuracy vs. Feature Count')
plt.savefig("interpretability and accuracy.png")
plt.show()
